function genId() { return Date.now().toString(36) + Math.random().toString(36).slice(2); }
function fmt(n) {
  if (n === null || n === undefined || n === '') return '―';
  const num = Number(n);
  const s = Math.abs(num).toLocaleString('ja-JP');
  return (num < 0 ? '−' : '') + '¥' + s;
}
function fmtDate(d) {
  if (!d) return '';
  const dt = new Date(d);
  return `${dt.getMonth()+1}/${dt.getDate()}`;
}
function ym(y, m) { return y * 100 + m; }
function entryYM(e) {
  const d = new Date(e.date);
  return ym(d.getFullYear(), d.getMonth() + 1);
}
function getEntriesForMonth(y, m) {
  return entries.filter(e => entryYM(e) === ym(y, m))
                .sort((a, b) => a.date.localeCompare(b.date));
}
function monthTotal(y, m) {
  return getEntriesForMonth(y, m).reduce((s, e) => s + Number(e.amount || 0), 0);
}
function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2000);
}
function freqLabel(f) {
  return {monthly:'毎月', bimonthly:'隔月', quarterly:'3ヶ月ごと', yearly:'年1回', 'semi-yearly':'半年ごと'}[f] || f;
}
function freqTag(f) {
  const cls = {monthly:'tag-monthly', bimonthly:'tag-bimonthly', quarterly:'tag-other', yearly:'tag-yearly', 'semi-yearly':'tag-other'}[f] || 'tag-other';
  return `<span class="tag ${cls}">${freqLabel(f)}</span>`;
}
function isPeriodicDue(p, y, m) {
  if (p.freq === 'monthly') return true;
  if (p.freq === 'bimonthly') return (m % 2 === p.month % 2);
  if (p.freq === 'quarterly') return (((m - p.month) % 3 === 0) && m >= p.month || (m + 12 - p.month) % 3 === 0);
  if (p.freq === 'yearly') return m === p.month;
  if (p.freq === 'semi-yearly') return (m === p.month || m === ((p.month + 5) % 12) + 1);
  return false;
}

  return all;
}

function addEntry() {
  const date = document.getElementById('in-date').value;
  const amount = document.getElementById('in-amount').value;
  const payee = document.getElementById('in-payee').value.trim();
  const purpose = document.getElementById('in-purpose').value.trim();
  const note = document.getElementById('in-note').value.trim();
  if (!date) { showToast('⚠ 日付を入力してください'); return; }
  if (!payee) { showToast('⚠ 支払先を入力してください'); return; }
  entries.push({ id: genId(), date, amount: Number(amount) || 0, payee, purpose, note });
  save(STORAGE_KEYS.entries, entries);

}

  save(STORAGE_KEYS.entries, entries);
  showToast(`✓ ${p.payee}を追加しました`);
  renderInputTab();
}

}

}
function deletePeriodic(id) {
  if (!confirm('この定期支払いを削除しますか？')) return;
  periodics = periodics.filter(p => p.id !== id);
  save(STORAGE_KEYS.periodic, periodics);
  showToast('🗑 削除しました');
  renderSettings();
}

function getAvailableYears() {
  const ys = entries.map(e => new Date(e.date).getFullYear());
  return [...new Set(ys)].sort((a, b) => b - a);
}
function renderYearly() {
  const years = getAvailableYears();
  if (!years.includes(selectedYear)) selectedYear = years[0] || new Date().getFullYear();

  // Year selector
  const ySel = document.getElementById('year-selector');
  ySel.innerHTML = years.map(y => `
    <button class="year-btn ${y === selectedYear ? 'active' : ''}" onclick="selectYear(${y})">${y}年</button>
  `).join('');

  // Payee aggregation
  const yearEntries = entries.filter(e => new Date(e.date).getFullYear() === selectedYear);
  const payeeMap = {};
  yearEntries.forEach(e => {
    if (!payeeMap[e.payee]) payeeMap[e.payee] = { total: 0, count: 0 };
    payeeMap[e.payee].total += Number(e.amount || 0);
    payeeMap[e.payee].count++;
  });
  const sorted = Object.entries(payeeMap).sort((a, b) => b[1].total - a[1].total);
  const grandTotal = sorted.reduce((s, [, v]) => s + v.total, 0);
  const payeeEl = document.getElementById('yearly-payee-list');
  if (sorted.length === 0) {
    payeeEl.innerHTML = '<div class="empty-state"><div class="empty-icon">📭</div><div class="empty-text">データがありません</div></div>';
  } else {
    payeeEl.innerHTML = sorted.map(([name, v]) => `
      <div class="payee-bar">
        <div>
          <div class="payee-name">${name}</div>
          <div class="payee-count">${v.count}回</div>
        </div>
        <div class="payee-total">${fmt(v.total)}</div>
      </div>
    `).join('') + `
      <div style="border-top:2px solid var(--primary);margin-top:8px;padding-top:8px;display:flex;justify-content:space-between;font-weight:700;font-size:15px">
        <span>年間合計</span>
        <span style="color:var(--primary)">${fmt(grandTotal)}</span>
      </div>
      <div style="text-align:right;font-size:12px;color:var(--accent2);margin-top:4px">
        請求分（50%）: ${fmt(Math.round(grandTotal * ratio))}
      </div>
    `;
  }

  // Monthly breakdown
  const tbody = document.getElementById('yearly-month-tbody');
  let rows = '';
  for (let mo = 1; mo <= 12; mo++) {
    const total = monthTotal(selectedYear, mo);
    const list = getEntriesForMonth(selectedYear, mo);
    rows += `<tr>
      <td>${mo}月</td>
      <td class="amount ${total < 0 ? 'negative' : ''}">${list.length > 0 ? fmt(total) : '―'}</td>
      <td class="amount">${list.length > 0 ? fmt(Math.round(total * ratio)) : '―'}</td>
    </tr>`;
  }
  tbody.innerHTML = rows;

  // Memo
  document.getElementById('yearly-memo').value = yearlyMemos[selectedYear] || '';
}
function selectYear(y) {
  selectedYear = y;
  renderYearly();
}
function saveYearlyMemo() {
  yearlyMemos[selectedYear] = document.getElementById('yearly-memo').value;
  save(STORAGE_KEYS.yearlyMemos, yearlyMemos);
  showToast('✓ メモを保存しました');
}
